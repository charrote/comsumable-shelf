    __tablename__ = "shelf_slot_events"

    id = Column(Integer, primary_key=True, autoincrement=True)
    shelf_slot_id = Column(Integer, ForeignKey("shelf_slots.id"), nullable=False)
    event_type = Column(String, nullable=False)       # occupied | released | error
    inventory_reel_id = Column(Integer, ForeignKey("inventory_reels.id"))
    source = Column(String, default="sensor")          # sensor | manual | api
    old_state = Column(Integer, default=0)             # 变化前的传感器状态
    new_state = Column(Integer, default=0)             # 变化后的传感器状态
    created_at = Column(DateTime, default=datetime.utcnow)
