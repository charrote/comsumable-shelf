## 一、当前状态 vs 目标状态

通过代码审查，核实了以下关键发现：

| 评估项 | 现状 | 重构目标 |
|--------|------|----------|
| **硬件协议** | Modbus TCP (FC2/FC15) via `app/hal/modbus.py` | HTTP REST JSON via `RackApiClient` |
| **LED 控制** | `LedService` → `Amkn8702g.set_led()` | `LedService` → `RackApiClient.light_up_cell()` |
| **储位轮询** | `SlotPollingService` → `SlotPoller` (3s Modbus FC2) | `RackSlotPoller` (10s HTTP GetCellList) + 回调 |
| **自动上架** | 传感器变化 → `_handle_change()` → 自动绑定 | 回调 → `rack_callback` → 自动绑定 |
| **WebSocket** | ❌ **不存在**（无推送服务） | ✅ 需要新建 WebSocket 服务 |
| **DB 迁移** | 手动 inline SQL（`init_db()` 中 `ALTER TABLE IF NOT EXISTS`） | 新增 Schema 变更 + 迁移脚本 |
| **前端料架页** | 只显示 Modbus 字段，无电池/控灯状态 | 新增控灯地址/电池预警/灯测试 |

### 需要特别关注的隐藏依赖

1. **WebSocket 服务不存在** — 任务清单 T10 提到"推送源从 SlotReader 改为 rack_callback"，但当前代码库中**没有任何 WebSocket 实现**。需要从零构建。
2. **`app/services/websocket_service.py` 和 `app/api/ws.py` 都不存在** — 需要新建。
3. **led_commands 表已存在**（在 `init_db()` 中创建），但字段不完整（缺少 `is_blink`/`turn_on_time`/`voice_text`），需要 migration。

#### T2 - RackApiClient 封装

**新建文件**: `backend/app/services/rack_api_client.py`

##### 核心设计

```python
"""
料架控灯 HTTP API 客户端。

封装所有与新智能料架的 HTTP REST 通信。
所有方法同步阻塞，由上层 async 服务在 executor 中调用。
"""

import uuid
import logging
import time
import requests
from typing import Optional, List

logger = logging.getLogger(__name__)


class RackApiError(Exception):
    """控灯 API 异常"""
    pass


class RackApiClient:
    """料架控灯 HTTP API 客户端"""

    # 色值映射（字符串 → integer）
    LED_COLORS = {
        "off": 0, "red": 1, "green": 2, "yellow": 3,
        "blue": 4, "magenta": 5, "cyan": 6, "white": 7,
    }
    # 警示灯色值映射
    INDICATOR_COLORS = {
        "off": 0, "red": 1, "yellow": 2, "red_yellow": 3,
        "green": 4, "red_green": 5, "yellow_green": 6, "all": 7,
    }

    def __init__(
        self,
        base_url: str,
        user_id: str = "",
        client_id: str = "",
        timeout: float = 5.0,
        max_retries: int = 3,
    ):
        self.base_url = base_url.rstrip("/")
        self.user_id = user_id
        self.client_id = client_id
        self.timeout = timeout
        self.max_retries = max_retries
        self.session = requests.Session()

    def _get_headers(self) -> dict:
        return {
            "userId": self.user_id,
            "clientId": self.client_id,
            "sessionId": str(uuid.uuid4()),
            "Content-Type": "application/json;charset=utf-8",
        }

    def _post(self, path: str, data: dict) -> dict:
        """通用 POST 请求（含重试）"""
        url = f"{self.base_url}{path}"
        last_error = None
        for attempt in range(self.max_retries):
            try:
                resp = self.session.post(
                    url, json=data, headers=self._get_headers(),
                    timeout=self.timeout,
                )
                resp.raise_for_status()
                result = resp.json()
                if result.get("code") != 0:
                    raise RackApiError(
                        f"API error: code={result.get('code')}, msg={result.get('message')}"
                    )
                return result
            except requests.Timeout as e:
                last_error = e
                if attempt < self.max_retries - 1:
                    sleep_time = 2 ** attempt  # 1s, 2s, 4s
                    logger.warning(
                        "RackApi timeout (attempt %d/%d), retrying in %ds",
                        attempt + 1, self.max_retries, sleep_time,
                    )
                    time.sleep(sleep_time)
            except requests.RequestException as e:
                last_error = e
                if attempt < self.max_retries - 1:
                    sleep_time = 2 ** attempt
                    time.sleep(sleep_time)
        raise RackApiError(f"Request failed after {self.max_retries} retries: {last_error}")

    # ── 公共 API ──

    def light_up_cell(
        self, cell_id: str, led_color: int, is_blink: bool = False, turn_on_time: int = 0
    ) -> dict:
        """单个储位亮灯"""
        return self._post("/api/RackCellMgr/LightUpCellLed", {
            "cellId": cell_id,
            "ledColor": led_color,
            "blink": is_blink,
            "turnOnTime": turn_on_time,
        })

    def light_up_cells_batch(
        self, cells: list, turn_on_time: int = 0, voice_text: str = ""
    ) -> dict:
        """批量亮灯 + 语音播报"""
        return self._post("/api/RackCellMgr/LightUpCellLedList", {
            "cells": cells,
            "turnOnTime": turn_on_time,
            "voiceText": voice_text,
        })

    def set_indicator_status(
        self, rack_id: str, indicator_id: int, indicator_status: int, is_blink: bool = False
    ) -> dict:
        """设置警示灯"""
        return self._post("/api/RackCellMgr/SetWarningLed", {
            "rackId": rack_id,
            "indicatorId": indicator_id,
            "indicatorStatus": indicator_status,
            "blink": is_blink,
        })

    def rack_test(self, rack_id: str, test_mode: int, interval: int = 0) -> dict:
         """硬件测试"""
         return self._post("api/RackMgr/RackTest", {
             "id": rack_id,
             "testMode": test_mode,
             "interval": interval,
         })

    def get_cell_list(
        self, rack_id: str = None, cell_filter: str = None,
        page_index: int = 1, page_size: int = 10,
    ) -> dict:
        """查询储位状态"""
        data = {"pageIndex": page_index, "pageSize": page_size}
        if rack_id:
            data["rackId"] = rack_id
        if cell_filter:
            data["filter"] = cell_filter
        return self._post("/api/RackCellMgr/GetCellList", data)
```

```python
"""
智能料架数据迁移脚本。

步骤:
  1. 备份 shelf_slots 表
  2. 根据 shelf.code + slot_on_board 生成 cell_id
  3. 回填 cell_id
  4. 验证迁移前后数量一致
  5. 删除 modbus_tcp_id、modbus_coil_base 列（可选）
"""

import asyncio
import logging
from sqlalchemy import text
from app.utils.database import engine

logger = logging.getLogger(__name__)


async def migrate():
    async with engine.begin() as conn:
        # 1. 备份
        await conn.execute(text("""
            CREATE TABLE IF NOT EXISTS shelf_slots_backup_20260626 AS
            SELECT * FROM shelf_slots;
        """))
        logger.info("Backup created: shelf_slots_backup_20260626")

        # 2. 查询 counts
        count_before = (await conn.execute(
            text("SELECT COUNT(*) FROM shelf_slots")
        )).scalar()

        # 3. 生成 cell_id: shelf.code(去特殊字符) + slot_on_board(4位)
        await conn.execute(text("""
            UPDATE shelf_slots
            SET cell_id = UPPER(REPLACE(REPLACE(s.code, '-', ''), ' ', ''))
                          || LPAD(CAST(sl.slot_on_board AS TEXT), 4, '0')
            FROM shelves s
            WHERE sl.shelf_id = s.id
              AND sl.cell_id IS NULL;
        """))
        logger.info("cell_id generated")

        # 4. 验证
        count_after = (await conn.execute(
            text("SELECT COUNT(*) FROM shelf_slots")
        )).scalar()
        null_cell_id = (await conn.execute(
            text("SELECT COUNT(*) FROM shelf_slots WHERE cell_id IS NULL")
        )).scalar()

        logger.info("Before: %d, After: %d, NULL cell_id: %d",
                     count_before, count_after, null_cell_id)

        if count_before != count_after:
            logger.error("Count mismatch! Rolling back...")
            # 从备份恢复逻辑
            return False

        # 5. 可选：删除旧字段
        # await conn.execute(text("""
        #     ALTER TABLE shelf_slots DROP COLUMN IF EXISTS modbus_tcp_id;
        # """))
        # await conn.execute(text("""
        #     ALTER TABLE shelf_slots DROP COLUMN IF EXISTS modbus_coil_base;
        # """))

        return True


if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO)
    success = asyncio.run(migrate())
    print("Migration success:", success)
```

```python
async def _process_queued_commands(self):
    """Process all queued LED commands from the database."""
    async with async_session_factory() as session:
        result = await session.execute(
            select(LedCommandModel)
            .where(LedCommandModel.status == "queued")
            .order_by(LedCommandModel.created_at.asc())
            .limit(20)
        )
        commands = result.scalars().all()

        for cmd in commands:
            try:
                if settings.HARDWARE_SIMULATION:
                    cmd.status = "sent"
                    cmd.sent_at = datetime.utcnow()
                    continue

                # 查 shelf + slot
                slot = await session.get(ShelfSlot, cmd.slot_id)
                if not slot:
                    cmd.status = "failed"
                    continue

                shelf = await session.get(Shelf, cmd.shelf_id)
                if not shelf or not shelf.rack_api_url:
                    cmd.status = "failed"
                    continue

                # 用 RackApiClient 替代 Modbus
                client = RackApiClient(
                    base_url=shelf.rack_api_url,
                    user_id=shelf.rack_api_userId or "",
                    client_id=shelf.rack_api_clientId or "",
                )

                color_int = RackApiClient.LED_COLORS.get(cmd.color, 2)
                client.light_up_cell(
                    cell_id=slot.cell_id,
                    led_color=color_int,
                    is_blink=cmd.is_blink or False,
                    turn_on_time=cmd.turn_on_time or 0,
                )

                cmd.status = "sent"
                cmd.sent_at = datetime.utcnow()

            except Exception as e:
                cmd.status = "failed"
                logger.error("led_command_failed", command_id=cmd.id, error=str(e))

        if commands:
            await session.commit()
```

#### T4 - 储位变化回调接口

**新建文件**: `backend/app/api/rack_callback.py`

```python
"""料架储位变化回调接口"""

import logging
from datetime import datetime
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from fastapi import APIRouter, Depends
from pydantic import BaseModel
from typing import List, Optional

from app.utils.database import get_db
from app.models import ShelfSlot, InventoryReel, ShelfSlotEvent

logger = logging.getLogger(__name__)
router = APIRouter(prefix="/rack", tags=["Rack Callback"])


class CellChangeItem(BaseModel):
    cellId: str
    status: int  # 0=取出, 1=放入
    timestamp: str


class CellChangeCallbackRequest(BaseModel):
    data: List[CellChangeItem]
    code: int = 0
    message: str = ""
    sessionId: str = ""


class CellChangeCallbackResponse(BaseModel):
    code: int = 0
    message: str = "OK"
    sessionId: str = ""


@router.post("/callback/cell-changed", response_model=CellChangeCallbackResponse)
async def cell_change_callback(
    data: CellChangeCallbackRequest,
    db: AsyncSession = Depends(get_db),
):
    """接收料架储位变化回调"""
    session_id = data.sessionId
    processed = 0

    for item in data.data:
        try:
            # 1. 查找 cell_id → shelf_slot
            result = await db.execute(
                select(ShelfSlot).where(ShelfSlot.cell_id == item.cellId)
            )
            slot = result.scalar_one_or_none()
            if not slot:
                logger.warning("Callback unknown cell_id: %s", item.cellId)
                continue

            bound_reel_id = None

            if item.status == 1:
                # ── 放入事件 → 自动绑定 ──
                reel = await _find_unbound_reel(db)
                if reel:
                    # 检查储位是否已被占用
                    existing = await db.execute(
                        select(InventoryReel).where(
                            InventoryReel.shelf_slot_id == slot.id,
                            InventoryReel.status == "on_shelf",
                        ).limit(1)
                    )
                    if not existing.scalar_one_or_none():
                        reel.shelf_slot_id = slot.id
                        reel.status = "on_shelf"
                        bound_reel_id = reel.id
                        logger.info(
                            "Auto-bound reel %d to slot %d (cell: %s)",
                            reel.id, slot.id, item.cellId,
                        )

            # 2. 记录事件
            event = ShelfSlotEvent(
                shelf_slot_id=slot.id,
                event_type="occupied" if item.status == 1 else "released",
                reel_id=bound_reel_id,
                source="api",
                old_state=abs(1 - item.status),
                new_state=item.status,
                cell_id=item.cellId,
                raw_data=item.model_dump(),
            )
            db.add(event)

            # 3. 更新 slot 状态
            slot.last_event_at = datetime.utcnow()
            slot.last_sensor_state = item.status

            processed += 1

        except Exception as e:
            logger.error("Callback processing error: %s", e, exc_info=True)

    await db.commit()

    # TODO: WebSocket 推送通知

    logger.info("Callback processed: %d/%d items", processed, len(data.data))
    return CellChangeCallbackResponse(
        sessionId=session_id,
        message=f"Processed {processed} items",
    )


async def _find_unbound_reel(db: AsyncSession) -> Optional[InventoryReel]:
    """查找最旧的待上架料盘"""
    result = await db.execute(
        select(InventoryReel)
        .where(
            InventoryReel.shelf_slot_id.is_(None),
            InventoryReel.status == "pending_shelving",
        )
        .order_by(InventoryReel.created_at.asc())
        .limit(1)
    )
    return result.scalar_one_or_none()
```

#### T5 - 储位状态查询改造

**新建文件**: `backend/app/services/rack_slot_poller.py`

```python
"""
RackSlotPoller — 替代旧 SlotReader 的 HTTP 轮询服务。

每 10s 调用 GetCellList 全量查询，更新 shelf_slots 表。
与回调结合：回调处理即时变化，轮询用于一致性校验。
"""

import asyncio
import logging
from datetime import datetime
from sqlalchemy import select, update

from app.config import settings
from app.models import Shelf, ShelfSlot
from app.services.rack_api_client import RackApiClient
from app.utils.database import async_session_factory

logger = logging.getLogger(__name__)


class RackSlotPoller:
    """料架储位 HTTP 轮询服务"""

    POLL_INTERVAL = 10  # 秒

    def __init__(self):
        self._running = False

    async def start(self):
        """启动轮询"""
        self._running = True
        logger.info("RackSlotPoller started (interval=%ds)", self.POLL_INTERVAL)
        while self._running:
            try:
                await self._poll_once()
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error("RackSlotPoller error: %s", e, exc_info=True)
            await asyncio.sleep(self.POLL_INTERVAL)

    async def stop(self):
        """停止轮询"""
        self._running = False
        logger.info("RackSlotPoller stopped")

    async def _poll_once(self):
        """单次轮询"""
        async with async_session_factory() as db:
            result = await db.execute(
                select(Shelf).where(Shelf.active == 1, Shelf.rack_api_url.isnot(None))
            )
            shelves = result.scalars().all()

            for shelf in shelves:
                try:
                    client = RackApiClient(
                        base_url=shelf.rack_api_url,
                        user_id=shelf.rack_api_userId or "",
                        client_id=shelf.rack_api_clientId or "",
                    )
                    await self._sync_shelf_cells(db, shelf.id, client)
                except Exception as e:
                    logger.warning("Poll shelf %d failed: %s", shelf.id, e)

    async def _sync_shelf_cells(self, db, shelf_id: int, client: RackApiClient):
        """同步单个料架的所有储位"""
        page_index = 1
        page_size = 100

        while True:
            resp = client.get_cell_list(
                rack_id=None, page_index=page_index, page_size=page_size
            )
            cells = resp.get("data", [])
            if not cells:
                break

            for cell in cells:
                cell_id = cell.get("cellId")
                if not cell_id:
                    continue

                # 更新 shelf_slots 表
                await db.execute(
                    update(ShelfSlot)
                    .where(
                        ShelfSlot.shelf_id == shelf_id,
                        ShelfSlot.cell_id == cell_id,
                    )
                    .values(
                        electricitys=cell.get("electricitys", 0),
                        last_sensor_state=1 if cell.get("used") else 0,
                        # led_id, rack_type 等可按需更新
                    )
                )

            if len(cells) < page_size:
                break
            page_index += 1

        await db.commit()
```

#### T10 - WebSocket 实时同步改造

##### 新建 `backend/app/services/websocket_service.py`

```python
"""
WebSocket 连接管理器。

管理所有 WebSocket 连接，提供广播能力。
支持储位变化推送和控灯状态推送。
"""

import json
import asyncio
import logging
from typing import Set
from fastapi import WebSocket

logger = logging.getLogger(__name__)


class WebSocketManager:
    """WebSocket 连接管理器"""

    def __init__(self):
        self._connections: Set[WebSocket] = set()

    async def connect(self, ws: WebSocket):
        await ws.accept()
        self._connections.add(ws)
        logger.info("WebSocket connected: %d total", len(self._connections))

    def disconnect(self, ws: WebSocket):
        self._connections.discard(ws)
        logger.info("WebSocket disconnected: %d remaining", len(self._connections))

    async def broadcast(self, event_type: str, data: dict):
        """向所有连接广播消息"""
        message = json.dumps({"type": event_type, "data": data}, ensure_ascii=False)
        dead: Set[WebSocket] = set()
        for ws in self._connections:
            try:
                await ws.send_text(message)
            except Exception:
                dead.add(ws)
        self._connections -= dead

    @property
    def connection_count(self) -> int:
        return len(self._connections)


# 全局单例
ws_manager = WebSocketManager()
```

##### 新建 `backend/app/api/ws.py`

```python
"""WebSocket 端点"""

import logging
from fastapi import APIRouter, WebSocket, WebSocketDisconnect
from app.services.websocket_service import ws_manager

logger = logging.getLogger(__name__)
router = APIRouter()


@router.websocket("/ws")
async def websocket_endpoint(ws: WebSocket):
    await ws_manager.connect(ws)
    try:
        while True:
            data = await ws.receive_text()
            if data == "ping":
                await ws.send_text("pong")
            # 可扩展其他消息类型
    except WebSocketDisconnect:
        ws_manager.disconnect(ws)
    except Exception as e:
        logger.warning("WebSocket error: %s", e)
        ws_manager.disconnect(ws)
```
