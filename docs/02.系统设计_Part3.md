### 4.2 ModbusTCPClient 设计

```python
class ModbusTCPClient:
    """
    AMKN8702G Modbus TCP 客户端
    
    功能:
    - TCP 长连接 (保持连接, 避免每次建连开销)
    - 自动重连 (指数退避)
    - 站号映射
    - 批量读取优化
    """
    
    def __init__(self, host: str, port: int = 502, timeout: float = 2.0):
        self.host = host
        self.port = port
        self.timeout = timeout
        self.controller_id = 200       # 主控板固定站号
        self._sync = AsyncModbusTcpClient(host, port)
        self._connected = False
        self._reconnect_attempts = 0
        
    async def connect(self) -> bool:
        """建立 TCP 长连接"""
        
    async def disconnect(self):
        """关闭连接"""
        
    def resolve_station(self, side: str, board_addr: int) -> tuple:
        """
        解析站号映射
        
        Returns:
            (modbus_tcp_id, com_channel)
            
        Examples:
            A面, 灯板1  → (1, 'COM1')
            A面, 灯板10 → (10, 'COM1')
            B面, 灯板1  → (64, 'COM2')  # 64 = 63 + 1
            B面, 灯板10 → (73, 'COM2')  # 73 = 63 + 10
        """
        if side == 'A':
            return (board_addr, 'COM1')
        else:
            return (63 + board_addr, 'COM2')
            
    async def read_coils(self, tcp_id: int, address: int, count: int) -> list:
        """读取线圈 (继电器控制状态)"""
        
    async def read_discrete_inputs(self, address: int, count: int) -> list:
        """
        读取数字量输入 (储位状态)
        
        A面: 地址 1000+
        B面: 地址 2000+
        """
        
    async def write_single_coil(self, tcp_id: int, address: int, value: bool):
        """写单线圈 (FC5, 单继电器/单LED)"""
        
    async def write_multiple_coils(self, tcp_id: int, address: int, values: list):
        """写多线圈 (FC15, 批量LED控制)"""
        
    async def read_holding_registers(self, address: int, count: int) -> list:
        """读保持寄存器 (配置信息, FC3)"""
        
    async def write_single_register(self, address: int, value: int):
        """写单寄存器 (FC6, 配置/命令)"""
        
    async def write_multiple_registers(self, address: int, values: list):
        """写多寄存器 (FC16, 批量配置)"""
        
    async def store_params(self):
        """存储参数: 60121 = 0x0001"""
        
    async def reset_device(self):
        """设备复位: 60121 = 0x0005"""
        
    async def execute_calibration(self):
        """执行校准: 60121 = 0x0004"""
        
    async def read_slot_status_batch(self, shelf: Shelf) -> dict:
        """
        批量读取某料架全部储位状态

        Args:
            shelf: Shelf 对象, 含 a_side_count, b_side_count
            
        Returns:
            {slot_id: {occupied: bool, last_seen: datetime}, ...}
        """
        # 从 system_settings 获取灯板储位数配置 (20/10/6)
        slots_per_board = int(get_setting('slot_capacity_config', '20'))

        # 计算需要读取的范围
        a_start = 1000
        b_start = 2000

        # A面批量读取
        if shelf.a_side_count > 0:
            a_count = shelf.a_side_count * slots_per_board
            a_bytes = read_discrete_inputs(self.controller_id, a_start, a_count)
            a_status = parse_coils_to_slots(a_bytes, a_count, slots_per_board, 'A')

        # B面批量读取
        if shelf.b_side_count > 0:
            b_count = shelf.b_side_count * slots_per_board
            b_bytes = read_discrete_inputs(self.controller_id, b_start, b_count)
            b_status = parse_coils_to_slots(b_bytes, b_count, slots_per_board, 'B')

        return {**a_status, **b_status}
        
    def start_keepalive(self, interval: int = 5):
        """启动心跳保持连接"""
```

### 4.3 SlotReader (储位轮询器) 设计

```python
class SlotReader:
    """
    储位状态实时轮询与变化检测
    
    职责:
    - 定时轮询所有料架的储位状态
    - 检测状态变化 (有物料 → 空 / 空 → 有物料)
    - 触发回调 (通知业务层)
    - 自动缓存与状态追踪
    """
    
    POLL_INTERVAL_SECONDS = 1.0       # 1秒轮询一次
    
    def __init__(self, gateway: ModbusTCPClient, db: Database):
        self.gateway = gateway
        self.db = db
        self._running = False
        self._last_state: dict[str, dict] = {}  # shelf_id → {slot_id: bool}
        self._callbacks = []
        self._change_history: deque = deque(maxlen=1000)
        
    async def start(self):
        """启动轮询循环"""
        self._running = True
        while self._running:
            try:
                await self._poll_all_shelves()
            except Exception as e:
                logger.error(f"SlotReader poll error: {e}")
            await asyncio.sleep(self.POLL_INTERVAL_SECONDS)
            
    def register_callback(self, callback: Callable[[SlotChange], None]):
        """注册状态变化回调"""
        self._callbacks.append(callback)
        
    async def _poll_all_shelves(self):
        """轮询所有活跃料架"""
        shelves = await self.db.get_active_shelves()
        
        for shelf in shelves:
            current_state = await self.gateway.read_slot_status_batch(shelf)
            prev_state = self._last_state.get(shelf.id, {})
            
            changes = self._detect_changes(current_state, prev_state)
            
            if changes:
                for change in changes:
                    await self._handle_change(change)
            
            self._last_state[shelf.id] = current_state
            
    def _detect_changes(self, current: dict, previous: dict) -> list:
        """
        对比前后状态, 识别变化
        
        Returns:
            [SlotChange(slot_id, material_id, from_occupied, to_occupied), ...]
        """
        changes = []
        all_slot_ids = set(current.keys()) | set(previous.keys())
        
        for slot_id in all_slot_ids:
            was_occupied = previous.get(slot_id, False)
            is_occupied = current.get(slot_id, False)
            
            if was_occupied != is_occupied:
                # 查找该储位对应的物料
                slot_info = self.db.get_slot_info(slot_id)
                material_id = self.db.get_material_in_slot(slot_id)
                
                changes.append(SlotChange(
                    slot_id=slot_id,
                    shelf_id=current[slot_id]['shelf_id'],
                    material_id=material_id,
                    was_occupied=was_occupied,
                    is_occupied=is_occupied
                ))
                
        return changes
        
    async def _handle_change(self, change: SlotChange):
        """处理储位变化"""
        # 记录到变化历史
        self._change_history.append(change)
        
        # 通知业务层回调
        for cb in self._callbacks:
            try:
                await cb(change)
            except Exception as e:
                logger.error(f"Callback error: {e}")
        
        # 自动触发逻辑:
        # - 空位变化 → 可能是已出库, 更新库存状态
        # - 有→空 → 触发出库确认
        # - 空→有 → 触发退库上架检测
```

### 4.4 LEDController 设计

```python
class LEDController:
    """
    三色灯控制 (协议2, 地址 10000+)
    
    核心机制:
    - 每个 LED 占 4 个线圈地址 (G + R + B + 控制位)
    - 控制位 = 1 表示本 LED 操作有效, = 0 表示保持
    - 支持局部更新 (只修改需要的 LED)
    """
    
    # 协议2 基地址
    LED_BASE_ADDRESS = 10000
    
    # 字节位映射 (协议2):
    # Byte 1: LED2控制位 | LED2B | LED2R | LED2G | LED1控制位 | LED1B | LED1R | LED1G
    
    @staticmethod
    def calculate_led_bytes(led_config: dict) -> bytes:
        """
        计算 LED 控制字节
        
        Args:
            led_config: {slot_id: {'color': 'green'|'red'|'blue', 'action': 'on'|'off'}, ...}
            
        Returns:
            字节串 (FC15 数据包)
        """
        # 初始化所有位为 0
        num_leds = max(slot_id for slot_id in led_config) if led_config else 0
        num_bytes = (num_leds * 4 + 7) // 8  # 每 LED 4 bits (G/R/B + ctrl)
        byte_data = bytearray(num_bytes)
        
        for slot_id, config in led_config.items():
            led_idx = slot_id - 1  # 0-based
            base_offset = led_idx * 4
            
            # 每 LED 占 4 bit: [控制位, B, R, G] (但实际映射见规格书)
            # 协议2: 每 2 LED = 1 字节
            
            led_group = led_idx // 2  # 0-based 组 (每字节 2 LEDs)
            led_in_group = led_idx % 2  # 0 = 奇数LED, 1 = 偶数LED
            
            byte_idx = led_group
            
            if led_in_group == 0:  # 奇数LED
                ctrl_bit = 3
                g_bit = 0
                r_bit = 1
                b_bit = 2
            else:  # 偶数LED
                ctrl_bit = 7
                g_bit = 4
                r_bit = 5
                b_bit = 6
                
            # 设置控制位
            byte_data[byte_idx] |= (1 << ctrl_bit)
            
            # 设置颜色
            color_map = {'green': g_bit, 'red': r_bit, 'blue': b_bit}
            if config['action'] == 'on':
                bit_pos = color_map.get(config['color'])
                if bit_pos is not None:
                    byte_data[byte_idx] |= (1 << bit_pos)
            # action='off' → 该颜色 bit = 0 (保持)
            
        return bytes(byte_data)
        
    async def set_led(self, shelf_id: int, slot_id: int, color: str, duration: int = 0):
        """
        设置单个储位 LED
        
        Args:
            shelf_id: 料架 ID
            slot_id: 储位全局序号
            color: green / red / blue
            duration: 亮灯秒数 (0=常亮, 5=亮5秒后灭)
        """
        slot = await self.db.get_slot_info(slot_id)
        tcp_id, _ = self.gateway.resolve_station(slot.side, slot.board_address)
        
        # 先读取当前所有 LED 状态 (保持其余不变)
        current_bytes = await self.gateway.read_discrete_inputs(
            tcp_id, self.LED_BASE_ADDRESS, slot.slot_on_board * 4
        )
        
        # 构建包含本 LED 的配置
        led_config = {slot_id: {'color': color, 'action': 'on'}}
        new_bytes = self.calculate_led_bytes(led_config)
        
        # 写入 (FC15)
        await self.gateway.write_multiple_coils(
            tcp_id,
            self.LED_BASE_ADDRESS,
            list(new_bytes)
        )
        
        # 保存命令记录
        await self.db.insert_led_command(
            shelf_id=shelf_id,
            slot_id=slot_id,
            color=color,
            duration=duration
        )
        
    async def clear_all_leds(self, shelf_id: int):
        """清除料架所有亮灯 (全部灭)"""
        
    async def clear_slot_led(self, slot_id: int):
        """清除单个储位亮灯"""
```

## 5. FIFO 算法与库存策略

### 5.1 核心策略

#### 5.1.1 尾数优先 (Tail-First)

```
规则: 优先出库剩余数量少的盘 (尾数盘)

示例:
需求: 100 盘 PCB板 A型
库存:
  Pallet-001:  30 盘 (尾数)
  Pallet-002:  50 盘
  Pallet-003:  40 盘
  Pallet-004: 100 盘 (满盘)

计算结果 (尾数优先):
  1. Pallet-001: 出库 30  (剩余 0 → 清空)
  2. Pallet-002: 出库 50  (剩余 0 → 清空)
  3. Pallet-003: 出库 20  (剩余 20 → 续用)
  
总出库: 100 盘, 选择 3 个盘
```

#### 5.1.2 时间优先 (Time-FIFO)

```
规则: 优先出库入库时间最早的盘

示例:
需求: 100 盘 PCB板 A型
库存 (按 last_in_time 排序):
  Pallet-003:  40 盘 (2026-06-10, 最早)
  Pallet-002:  50 盘 (2026-06-11)
  Pallet-001:  30 盘 (2026-06-12)
  Pallet-004: 100 盘 (2026-06-15, 最新)

计算结果 (时间优先):
  1. Pallet-003: 出库 40  (剩余 0 → 清空)
  2. Pallet-002: 出库 50  (剩余 0 → 清空)
  3. Pallet-001: 出库 10  (剩余 20 → 续用)
  
总出库: 100 盘, 选择 3 个盘
```

#### 5.1.3 混合策略 (可配置)

```
规则: 同尾数时, 时间优先
即: 先按剩余量排序 (尾数优先), 尾数相同时按入库时间排序

示例:
需求: 50 盘
库存:
  Pallet-A:  30 盘 (2026-06-10, 最早)
  Pallet-B:  30 盘 (2026-06-12, 最新)
  Pallet-C:  80 盘 (2026-06-08, 最早)

计算结果 (混合):
  1. Pallet-A: 出库 30  (尾数优先)
  2. Pallet-B: 出库 20  (尾数次优先)
  
总出库: 50 盘, 选择 2 个盘
```

### 5.2 FIFO 算法伪代码

```python
def calculate_fifo_pallets(
    material_id: int,
    required_qty: float,
    strategy: str,
    db: Database
) -> list[dict]:
    """
    计算出库库存盘序列
    
    Args:
        material_id: 物料 ID
        required_qty: 需求数量
        strategy: 'tail_first' / 'time_fifo' / 'mixed'
        db: 数据库连接
        
    Returns:
        [
            {
                'pallet_id': 5001,
                'quantity': 30.0,
                'last_in_time': '...',
                'shelf_slot_id': 42,
                'reason': 'tail_first_priority'
            },
            ...
        ]
    """
    # 1. 获取所有 on_shelf 状态的库存盘
    on_shelf_pallets = db.query("""
        SELECT id, quantity, original_quantity, last_in_time, 
               shelf_slot_id, status, customer_id
        FROM inventory_pallets
        WHERE material_id = ? AND status = 'on_shelf'
          AND quantity > 0
        ORDER BY last_in_time ASC
    """, (material_id,))
    
    if not on_shelf_pallets:
        return []  # 无可用库存
    
    # 2. 按策略排序 (系统配置可切换 tail_first / time_fifo)
    fifo_strategy = strategy  # 'config' 时从 system_settings 读取
    if fifo_strategy == 'config':
        fifo_strategy = get_setting('fifo_strategy', 'tail_first')
    
    if fifo_strategy == 'tail_first':
        on_shelf_pallets.sort(key=lambda p: p['quantity'])
    elif strategy == 'time_fifo':
        on_shelf_pallets.sort(key=lambda p: p['last_in_time'])
    elif strategy == 'mixed':
        on_shelf_pallets.sort(key=lambda p: (p['quantity'], p['last_in_time']))
    
    # 3. 贪心选择
    selected = []
    remaining = required_qty
    
    for pallet in on_shelf_pallets:
        if remaining <= 0:
            break
            
        take_qty = min(pallet['quantity'], remaining)
        selected.append({
            'pallet_id': pallet['id'],
            'quantity': take_qty,
            'last_in_time': pallet['last_in_time'],
            'shelf_slot_id': pallet['shelf_slot_id'],
            'remaining_after': pallet['quantity'] - take_qty
        })
        remaining -= take_qty
    
    # 4. 检查是否满足
    total_selected = sum(s['quantity'] for s in selected)
    shortage = required_qty - total_selected
    
    return {
        'pallets': selected,
        'total_selected': total_selected,
        'shortage': shortage,
        'strategy_used': strategy
    }
```

### 5.3 退库 + XR 配对逻辑

```python
async def handle_xr_match(
    xr_batch: dict,           # 从 XR 设备上传
    tracking_pallets: list    # 当前追踪中的库存盘
) -> dict:
    """
    XR 点料机数据配对
    
    逻辑:
    1. 在 tracking 状态的库存盘中查找匹配
    2. 匹配条件:
       - 物料编码相同
       - 时间差在 ±5s 窗口内 (可配置)
    3. 配对成功 → 标记为 matched
    4. 配对失败 → 标记为 failed, 需人工处理
    """
    # 1. 查找同物料 + 时间窗口内的追踪盘
    matches = db.query("""
        SELECT p.*, t.last_out_time
        FROM inventory_pallets p
        JOIN transactions t ON t.inventory_pallet_id = p.id
        WHERE p.customer_id = ?
          AND t.source_type = 'issue'
          AND p.status = 'tracking'
          AND p.quantity > 0
        ORDER BY t.last_out_time DESC
    """, (xr_batch['customer_id'],))
    
    # 2. 在 ±window 秒内找最近一次出库
    window = get_setting('xr_match_window_seconds', 5)
    best_match = None
    best_delta = float('inf')
    
    for pallet in matches:
        delta = abs(
            (datetime.now() - pallet['last_out_time']).total_seconds()
        )
        if delta <= window and delta < best_delta:
            best_delta = delta
            best_match = pallet
    
    if best_match:
        # 配对成功
        db.execute("""
            UPDATE inventory_pallets 
            SET status = 'ready_restock',
                xr_matched = 1,
                xr_match_key = ?,
                xr_counted_qty = ?
            WHERE id = ?
        """, (xr_batch['match_key'], xr_batch['counted_qty'], best_match['id']))
        
        return {
            'status': 'matched',
            'pallet_id': best_match['id'],
            'time_delta': best_delta,
            'action': 'confirm_restock'  # 下一步: 确认退库 + 上架
        }
    else:
        # 未找到配对
        return {
            'status': 'unmatched',
            'action': 'manual_review'  # 需要人工介入
        }
```

### 5.4 防呆逻辑

```python
async def validate_scan(
    barcode: str,
    scan_type: str,     # 'inbound' | 'outbound'
    order_id: int,
    operator_id: int
) -> dict:
    """
    扫码校验与防呆
    
    场景:
    - 入库: 校验物料编码有效性, 是否重复扫码
    - 出库: 校验是否在发料单中, 是否已出库, 数量匹配
    """
    # 1. 解析条码
    parsed = parse_barcode(barcode)
    if not parsed:
        return {'status': 'error', 'message': '无效的条码格式'}
    
    material = get_material(parsed['code'])
    if not material:
        return {'status': 'error', 'message': f'物料 {parsed["code"]} 不存在'}
    
    if scan_type == 'inbound':
        # 入库防呆
        existing = db.query("""
            SELECT id, quantity FROM inventory_pallets
            WHERE material_id = ? AND customer_code = ?
              AND status IN ('on_shelf', 'tracking')
        """, (material['id'], parsed['customer_code']))
        
        if existing:
            # 重复扫码
            return {
                'status': 'duplicate',
                'matched_pallet_id': existing['id'],
                'action': 'confirm_merge'  # 主管授权合并
            }
        
        return {'status': 'ok', 'action': 'create_receipt'}
    
    elif scan_type == 'outbound':
        # 出库防呆
        order_detail = db.query("""
            SELECT id, required_qty, picked_qty, pallet_ids
            FROM issue_detail
            WHERE issue_order_id = ? AND material_id = ?
              AND status IN ('pending', 'picking')
        """, (order_id, material['id']))
        
        if not order_detail:
            return {
                'status': 'error',
                'message': f'物料 {parsed["code"]} 不在发料单 #{order_id} 中'
            }
        
        # 检查是否已在该单中出库
        if parsed['pallet_id'] in order_detail['pallet_ids']:
            return {
                'status': 'duplicate_out',
                'message': '该物料盘已出库, 禁止重复操作'
            }
        
        # 检查是否已完全出库
        if order_detail['picked_qty'] >= order_detail['required_qty']:
            return {
                'status': 'completed',
                'message': '该物料需求已全部出库'
            }
        
        return {'status': 'ok', 'action': 'confirm_pick'}
```
